import React, { Component } from "react";
//import YouTube from 'react-youtube';
import ReactPlayer from "react-player";

/**
 * VideoCard Component
 * 
 * A React component that renders a video player using the ReactPlayer library. 
 * It takes a video URL as a prop and displays it in a responsive container.
 * 
 * @extends Component
 */
class VideoCard extends Component {
  state = {};
  render() {
    return (
      <div className="player-wrapper w-100 position-relative ">
        {/* <h5 className="card-title p-3">{this.props.title}</h5>  */}
        <div className="video-container  d-flex justify-content-center align-items-center">
          <ReactPlayer
            controls={true}
            url={this.props.url}
            playsInline={true}
            height={"100%"}
            width={"100%"}
          />
        </div>
      </div>
    );
  }

  _onReady(event) {
    // access to player in all event handlers via event.target
    event.target.pauseVideo();
  }
}

export default VideoCard;
