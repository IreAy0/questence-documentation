import React, { useState } from "react";
import { Collapse } from "react-bootstrap";

/**
 * BookmarkCard component
 *
 * A React component that displays a bookmark card with a collapsible notes section.
 *
 * @param {object} props - Component props
 * @param {string} props.lessonTopic - The topic of the lesson
 * @param {string} props.datePosted - The date the bookmark was posted
 * @param {string} props.url - The URL of the bookmark
 * @param {string} props.notes - The notes associated with the bookmark
 *
 * @example
 * <BookmarkCard
 *   lessonTopic="React Fundamentals"
 *   datePosted="2022-01-01"
 *   url="https://example.com/react-fundamentals"
 *   notes="This is a great resource for learning React!"
 * />
 */
const BookmarkCard = (props) => {
  const [open, setOpen] = useState(false);
  const { lessonTopic, datePosted, url, notes } = props;

  return (
    <>
      <div className="card mb-3">
        <div className="card-body pb-1">
          <div className="card-title fw-bold">
            <a href={url}>{lessonTopic}</a>
            <small className="text-muted float-end">
              {datePosted}{" "}
              <i
                onClick={() => setOpen(!open)}
                className="bi bi-chevron-down"
              ></i>
            </small>
          </div>
        </div>
        <Collapse in={open}>
          <div className="card-footer">
            <i>{notes}</i>
          </div>
        </Collapse>
      </div>
    </>
  );
};

export default BookmarkCard;
