import React, { Component } from "react";

/**
 * ImageCard component that displays an image based on the provided props.
 *
 * @class ImageCard
 * @extends Component
 */

class ImageCard extends Component {
  state = {};

   /**
   * Renders the ImageCard component.
   *
   * @returns {JSX.Element} The rendered component.
   */
  render() {
    let style = {};

    switch (this.props.position) {
      case "BA":
        style = {
          maxWidth: "100%",
          with: "100%",
          float: "center",
        };
        break;

      case "RI":
        style = {
          maxWidth: "100%",
          float: "right",
        };
        break;

      case "CE":
        style = {
          maxWidth: "100%",
          float: "center",
        };
        break;

      default:
        style = {
          maxWidth: "100%",
          float: "left",
        };
        break;
    }

    return (
      <div className="p-0">
        <img src={this.props.content} alt={this.props.title} style={style} className="rounded-3" />
      </div>
    );
  }
}

/**
 * @typedef {Object} ImageCardProps
 * @property {string} content - The source of the image to display.
 * @property {string} title - The alt text for the image.
 * @property {string} position - The position of the image ('BA', 'RI', 'CE', or any other).
 */

export default ImageCard;
