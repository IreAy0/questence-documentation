import React from "react";
// import Lottie from "react-lottie";
import Lottie from "lottie-react";
import animationData from "../animations/loader.json";

/**
 * A loading indicator component that displays an animated spinner when visible is true.
 * By default, the loading indicator is not visible.
 *
 * @param {boolean} visible - Whether the loading indicator should be visible or not. Default is false.
 * @example
 * import React from 'react';
 * import ActivityIndicator from './ActivityIndicator';
 *
 * function MyComponent() {
 *   const [loading, setLoading] = React.useState(false);
 *
 *   function handleClick() {
 *     setLoading(true);
 *     // Simulate a delay with setTimeout
 *     setTimeout(() => setLoading(false), 2000);
 *   }
 *
 *   return (
 *     <div>
 *       <button onClick={handleClick}>Click me!</button>
 *       <ActivityIndicator visible={loading} />
 *     </div>
 *   );
 * }
 */
const ActivityIndicator = ({ visible = false }) => {
  if (!visible) return null;
  return (
    <div
      style={{
        position: "fixed",
        // top: "200px",
        // zIndex: 1,
        // right: "30%",
        background: "rgba(0, 0, 0, 0.6)",
        // height: "300px",
        // width: "300px",
        width: "100%",
        height: "100%",
        top: 0,
        left: 0,
        right: 0,
        bottom: 0,
        backgroundColor: "rgba(0, 0, 0, 0.5)",
        zIndex: 1053,
        cursor: "pointer",
      }}
    >
      <div
        style={{
          position: "absolute",
          top: "50%",
          left: "50%",
          transform: "translate(-50%, -50%)",
        }}
      >
        {/* <Lottie options={defaultOptions} height={300} width={300} /> */}
        <div
          style={{
            height: 300,
            width: 300,
          }}
        >
          <Lottie animationData={animationData} loop={true} />
        </div>
      </div>
    </div>
  );
};

export default ActivityIndicator;
